while height < goalHeight
    [height, velocity, acceleration, direction] = readSensorData()
    yVelocity = getYComponent(velocity, direction)
    projectedAccel = -1*yVelocity^2 / 2*(goalHeight - height)
    neededAccel = projectedAccel - acceleration
    neededForce = neededAccel * dryMass
    deploymentAngle = calcNewDrag(neededForce)
    movePaddleArms(deploymentAngle)

calcNewDrag(neededForce)
